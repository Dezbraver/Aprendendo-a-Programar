int invert( void )
{
  int n;
  int i;
  int invert = 0;
  int pos = 1;

  scanf( "%d", &n );

  for ( i = 1; n / (i * 10); i *= 10 );

  for ( ; i >= 1; i /= 10 )
  {
    invert += ( n / i ) * pos;
    n %= i;
    pos *= 10;
  } /* end (for) */

  return invert;
} /* end [invert] */
