int qualityPoints( int media )
{
  if ( media >= 90 && media <= 100 )
    return 4;
  if ( media >= 80 && media <= 89 )
    return 3;
  if ( media >= 70 && media <= 79 )
    return 2;
  if ( media >= 60 && media <= 69 )
    return 1;
  if ( media <= 59 )
    return 0;
} /* end [qualityPoints] */
